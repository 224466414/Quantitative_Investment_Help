计算 MKT/SMB/HML

# -*- encoding: utf-8 -*-
'''
Filename         :东方证券_009_MKT-SMB-HML.py
Description      :该模块用于计算因子MKT/SMB/HML
Author           :量化投资1组
Version          :1.0
'''

import numpy as np
import pandas as pd
import time, sys
from collections import OrderedDict
from concurrent.futures import ProcessPoolExecutor
sys.path.append('modules')
from getStock import stockclass
from getDate import dateclass
from getQuote import quoteclass
import basicFunction as fun

# 设定开始与结束日期
beginDate = '20050101'
endDate = '20210531'

# 创建日期数据获取对象
dateClass = dateclass()
dates = dateClass.get_dates(beginDate, endDate, showProcess=False)

# 创建股票日频数据获取对象
stockClass = stockclass('data/stock/1day', preRead=['MVFloat', 'PB'], showProcess=False)
stockPools = stockClass.get_stockPools(dates, showProcess=False)

# 创建股票分钟频数据获取对象
quoteClass = quoteclass('data/stock/5min')

funStart = 0


# 定义计算主体
def get_results(thisDate):
    # 设置有序字典以存储当日的因子
    factorMKTDate, factorSMBDate, factorHMLDate = OrderedDict(), OrderedDict(), OrderedDict()
    # 获取上一日与下一日的日期
    lastDate = dateClass.get_beforeDate(thisDate, 1, showProcess=False)
    # 如果上一日有数据
    if lastDate != 'out of range':
        # 获取上一日权重数据
        dataOriginal = stockClass.get_data(secIDs=stockPools[thisDate], dates=[lastDate], fields=['PB', 'MVFloat'], showProcess=False)
        # 计算MKT权重
        dataLogMVFloat = np.log(dataOriginal['MVFloat'])
        weightsMKT = dataLogMVFloat / dataLogMVFloat.sum()
        # 计算SMB权重
        dataMVFloatSmall = dataOriginal['MVFloat'].sort_values(ascending=True).iloc[: int(len(dataOriginal) / 3)]
        dataMVFloatBig = dataOriginal['MVFloat'].sort_values(ascending=False).iloc[: int(len(dataOriginal) / 3)]
        dataLogMVFloatSmall = np.log(dataMVFloatSmall)
        dataLogMVFloatBig = np.log(dataMVFloatBig)
        weightsSmall = dataLogMVFloatSmall / dataLogMVFloatSmall.sum()
        weightsBig = dataLogMVFloatBig / dataLogMVFloatBig.sum()
        # 计算HML权重
        indexHigh = dataOriginal['PB'].sort_values(ascending=True).iloc[: int(len(dataOriginal['PB']) / 3)].index.tolist()
        indexLow = dataOriginal['PB'].sort_values(ascending=False).iloc[: int(len(dataOriginal['PB']) / 3)].index.tolist()
        dataMVFloatHigh = dataOriginal.loc[indexHigh, 'MVFloat']
        dataMVFloatLow = dataOriginal.loc[indexLow, 'MVFloat']
        dataLogMVFloatHigh = np.log(dataMVFloatHigh)
        dataLogMVFloatLow = np.log(dataMVFloatLow)
        weightsHigh = dataLogMVFloatHigh / dataLogMVFloatHigh.sum()
        weightsLow = dataLogMVFloatLow / dataLogMVFloatLow.sum()

        # 获取本日价格数据
        dataPrice = quoteClass.get_data(secIDs=stockPools[thisDate], dates=[thisDate], fields=['Open', 'Close'], showProcess=False)
        # 遍历所有K线
        for order in range(0, 48):
            if order == 0:
                thisPrice = dataPrice[dataPrice['order'] == order]['Open']
                nextPrice = dataPrice[dataPrice['order'] == order]['Close']
                RetOrder = nextPrice / thisPrice - 1
            else:
                lastPrice = dataPrice[dataPrice['order'] == order - 1]['Close']
                thisPrice = dataPrice[dataPrice['order'] == order]['Close']
                RetOrder = thisPrice / lastPrice - 1
            # 计算MKT数据
            MKTIndex = fun.extract_samePart(RetOrder.index.tolist(), weightsMKT.index.tolist())
            RetOrderMKT = RetOrder[MKTIndex]
            factorMKTDate[order] = (RetOrderMKT * weightsMKT).sum()
            # 计算SMB数据
            SMBSmallIndex = fun.extract_samePart(RetOrder.index.tolist(), weightsSmall.index.tolist())
            SMBBigIndex = fun.extract_samePart(RetOrder.index.tolist(), weightsBig.index.tolist())
            RetOrderSmall, RetOrderBig = RetOrder[SMBSmallIndex], RetOrder[SMBBigIndex]
            factorSMBDate[order] = (RetOrderSmall * weightsSmall).sum() - (RetOrderBig * weightsBig).sum()
            # 计算HML数据s
            HMLHighIndex = fun.extract_samePart(RetOrder.index.tolist(), weightsHigh.index.tolist())
            HMLLowIndex = fun.extract_samePart(RetOrder.index.tolist(), weightsLow.index.tolist())
            RetOrderHigh, RetOrderLow = RetOrder[HMLHighIndex], RetOrder[HMLLowIndex]
            factorHMLDate[order] = (RetOrderHigh * weightsHigh).sum() - (RetOrderLow * weightsLow).sum()
        # 整理数据
        factorMKTDate, factorSMBDate, factorHMLDate = pd.Series(factorMKTDate), pd.Series(factorSMBDate), pd.Series(factorHMLDate)

    # 如果上一日无数据
    else:
        factorMKTDate = pd.Series({order: np.nan for order in range(0, 48)})
        factorSMBDate = pd.Series({order: np.nan for order in range(0, 48)})
        factorHMLDate = pd.Series({order: np.nan for order in range(0, 48)})

    factorResult = {'MKT': factorMKTDate, 'SMB': factorSMBDate, 'HML': factorHMLDate}

    # 显示当前进度
    index = dates[dates == thisDate].index.tolist()[0]
    funcPerc = len(dates[: index]) / len(dates)
    fun.processing_bar(funcPerc, funStart)

    return factorResult


if __name__ == '__main__':
    # 计算程序运行时间
    print('\n因子计算开始')
    funStart = time.perf_counter()

    # 设置数据框存储因子
    factorMKTResult = pd.DataFrame(columns=range(0, 48))
    factorSMBResult = pd.DataFrame(columns=range(0, 48))
    factorHMLResult = pd.DataFrame(columns=range(0, 48))

    # 使用多进程并发计算
    with ProcessPoolExecutor() as pool:
        results = pool.map(get_results, dates)
        results = list(zip(dates, results))

    # 整理计算结果
    for date, result in results:
        factorMKTResult.loc[date, :] = result['MKT']
        factorSMBResult.loc[date, :] = result['SMB']
        factorHMLResult.loc[date, :] = result['HML']

    print('\n因子计算完成')

    # 给index加上名称
    factorMKTResult.index = pd.Series(factorMKTResult.index, name='dates')
    factorSMBResult.index = pd.Series(factorMKTResult.index, name='dates')
    factorHMLResult.index = pd.Series(factorMKTResult.index, name='dates')

    # 输出计算结果
    print(factorMKTResult)
    print(factorSMBResult)
    print(factorHMLResult)

    # 存储计算结果
    factorMKTResult.to_csv('factor/东方证券_009/东方证券_009_MKT.csv')
    factorSMBResult.to_csv('factor/东方证券_009/东方证券_009_SMB.csv')
    factorHMLResult.to_csv('factor/东方证券_009/东方证券_009_HML.csv')
